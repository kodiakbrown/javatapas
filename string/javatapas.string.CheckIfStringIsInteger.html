Java: Check If String Is Integer


// Title: Check If String Is Integer
// Keywords: java/string/integer

package javatapas.string;

/**
* Check if the value of a String variable is an Integer value.
* @author <a href="https://stackoverflow.com/questions/5439529/determine-if-a-string-is-an-integer-in-java">Stack Overflow Thread</a>
* @version 2020-05-01
*/

public class CheckIfStringIsInteger {

	/**
	* @param inStr The string to be evaluated as an integer.
	*/

	public static boolean isInteger(String inStr) {
	    return isInteger(inStr,10);
	}

	/**
	* @param inStr The string to be evaluated as an integer.
	* @param radix Indicates the numeral system to use (default is Base 10).
	*/

	public static boolean isInteger(String inStr, int radix) {
	    if(inStr.isEmpty()) return false;
	    for(int i = 0; i < inStr.length(); i++) {
	        if(i == 0 && inStr.charAt(i) == '-') {
	            if(inStr.length() == 1) return false;
	            else continue;
	        }
	        if(Character.digit(inStr.charAt(i),radix) < 0) return false;
	    }
	    return true;
	}

} //
