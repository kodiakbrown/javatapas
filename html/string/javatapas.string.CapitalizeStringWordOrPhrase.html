Java: Capitalize String Word Or Phrase


// Title: Capitalize String Word Or Phrase
// Keywords: java/string/capitalize/lowercase/uppercase

package javatapas.string;

/**
* Capitalize the value of Word or a Phrase.
* @author Kodiak Brown, LLC published on <a href="https://www.javatapas.com">https://www.javatapas.com</a>
* @version 2020-05-01
*/

public class CapitalizeStringWordOrPhrase {

	/**
	* @param inStr The word or phrase to be capitalized.
	*/

	public static String capitalize(String inStr) {
		if (inStr.contains(" ")){return capitalizePhrase(inStr);}
		else {return capitalizeWord(inStr);}
	}

	/**
	* @param inStr The single word to be capitalized.
	*/

	public static String capitalizeWord(String inStr) {
		if (inStr == null || inStr.length() == 0) return inStr;
		return inStr.substring(0, 1).toUpperCase() + inStr.substring(1).toLowerCase();
	}

	/**
	* @param inStr The phrase or multi-part word to be capitalized.
	*/

	public static String capitalizePhrase(String inStr) {

		if (inStr == null || inStr.trim().length() == 0) return inStr;

		StringBuilder sb = new StringBuilder();
		String[] words = inStr.split("\\s");
	    for (String s : words) {sb.append(capitalizeWord(s) + " ");}
		sb.delete(sb.length() - 1, sb.length());

		return sb.toString();

	}

} //
