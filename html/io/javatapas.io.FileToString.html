// Title: File To String

package javatapas.io;

import java.io.*;

/**
* Instantiate a Java String from the contents in a Text file.
* @author Kodiak Brown, LLC published on <a href="https://www.javatapas.com">https://www.javatapas.com</a>
* @version 2020-04-28
*/

public class FileToString {

/**
* @param fileName The name of the source file.
*/

public static String getString(String fileName) throws IOException {

	StringBuilder sb = new StringBuilder();
	BufferedReader reader = null;

	try {
		reader = new BufferedReader(new FileReader(fileName));
		char[] buf = new char[1024];
		int numRead = 0;
		String readData;
		while ((numRead = reader.read(buf)) != -1) {sb.append(buf, 0, numRead);}
	}

	finally {
		try {reader.close();}
		catch (IOException ex) {ex.printStackTrace();}
	}

	return sb.toString();
}

} //
