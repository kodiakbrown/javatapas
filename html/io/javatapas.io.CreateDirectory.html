Java: Create Directory


// Title: Create Directory
// Keywords: java/file/mkdir/directory

package javatapas.io;

import java.io.File;

/**
* Create a new file system directory.
* @author Kodiak Brown, LLC published on <a href="https://www.javatapas.com">https://www.javatapas.com</a>
* @version 2020-05-02
*/

public class CreateDirectory {

/**
* @param directoryName The name of the new directory to be created.
*/

public static void mkdirs (String directoryName){

	File directory = new File(directoryName);

	if (!directory.exists()){directory.mkdirs();}

}

} //
